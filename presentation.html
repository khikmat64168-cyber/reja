03 server.js — Backend kodi
const express = require('express'); const fs = require('fs'); const app = express(); // 1. Public papkadagi fayllarni (CSS, rasm) brauzerga ochish app.use(express.static('public')); // 2. EJS shablon dvigatelini ulash app.set('view engine', 'ejs'); app.set('views', 'views'); // 3. user.json faylini o'qish let user; fs.readFile('database/user.json', 'utf-8', (err, data) => { if (!err) user = JSON.parse(data); }); // 4. /author manziliga so'rov kelganda author.ejs ni render qilish app.get('/author', (req, res) => { res.render('author', { user: user }); }); app.listen(3000);

res.render() nima qiladi?

author.ejs faylini ochib, ichidagi <%= user.name %> kabi joylarga ma'lumot qo'yib, tayyor HTML ni brauzerga yuboradi.

fs.readFile() nima qiladi?

user.json faylini diskdan o'qib, JSON formatdan JavaScript obyektiga o'giradi. Natijani user o'zgaruvchisiga saqlaydi.

06 EJS — Shablon sintaksisi

<%= %> — Qiymat chiqarish

O'zgaruvchi qiymatini HTML ga chiqaradi.

<%= user.name %> → "Matthew"

<% %> — Kod yozish

JavaScript kodi yozadi, lekin ekranga chiqarmaydi.

<% if (user.age > 18) { %> <p>Voyaga yetgan</p> <% } %>

forEach — Massiv aylantirish

JSON massividagi har bir element uchun HTML yaratadi.

<% skills.forEach(s => { %> <div><%= s.name %></div> <% }) %>
Misol: user.json dagi "languages": [{"name":"Uzbek","percent":100}] → EJS forEach bilan har biri uchun progress bar chiqaradi → brauzerda Uzbek — 100% ko'rinadi.
// user.json { "name": "Matthew", "skills": [{ "name": "Html", "percent": 90 }] } // server.js res.render('author', { user: user }); ← user ni EJS ga yuboradi // author.ejs <%= user.name %> → "Matthew" bo'lib chiqadi
07 main.css — CSS Flexbox bilan layout
/* Tashqi fon — markazga joylashtirish */ .frame { display: flex; justify-content: center; /* gorizontal markaz */ background-color: #E8E8E8; } /* Asosiy konteyner */ .main_frame { display: flex; flex-direction: row; /* gorizontal */ width: 1200px; } /* Chap panel — fixed kenglik */ .left_frame { width: 305px; flex-shrink: 0; /* kichraymaydi */ flex-direction: column; /* vertikal */ } /* O'rta qism — qolgan joyni oladi */ .mid_frame { flex: 1; /* bo'sh joyni to'ldiradi */ }

display: flex nima?

Farzandlarni bir qatorda (row) yoki ustun (column) qilib joylashtiradi. Zamonaviy layout usuli.

flex: 1 nima?

mid_frame — left_frame (305px) va right_frame (108px) dan keyin qolgan barcha bo'sh joyni egallaydi.

flex-shrink: 0 nima?

Ekran kichrayganda ham left_frame ning kengligi 305px dan kichraymaydi.
